def evaluate_model(y_true, y_pred, name):
    print(f"\n{name} Results:")
    print("R Score:", round(r2_score(y_true, y_pred), 3))
    print("MAE:", round(mean_absolute_error(y_true, y_pred), 3))
    print("RMSE:", round(np.sqrt(mean_squared_error(y_true, y_pred)), 3))

evaluate_model(y_test, y_pred_lr, "Linear Regression")

# Test stationarity using ADF test
from statsmodels.tsa.stattools import adfuller # Import the adfuller function

def test_stationarity(timeseries):
    result = adfuller(timeseries.dropna())
    print(f'ADF Statistic: {result[0]}')
    print(f'p-value: {result[1]}')
    for key, value in result[4].items():
        print(f'{key}: {value}')

from statsmodels.tsa.arima.model import ARIMA

# Fit ARIMA model (example: ARIMA(1, 1, 1))
model = ARIMA(time_series_data, order=(1, 1, 1))
model_fit = model.fit()

# Forecast next 5 years
forecast = model_fit.forecast(steps=5)

# Create future year index
last_year = time_series_data.index.max()
future_years = pd.RangeIndex(start=last_year + 1, stop=last_year + 6)

# Assign future index
forecast.index = future_years

# Plot
plt.figure(figsize=(10, 5))
time_series_data.plot(label='Historical', marker='o')
forecast.plot(label='Forecast', linestyle='--', marker='x', color='orange')
plt.title("Forecast of Average Overnight Shelter Stays (Next 5 Years)", fontsize=14)
